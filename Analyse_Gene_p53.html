from Bio.Blast import NCBIXML
from Bio import SeqIO

from Bio.Align.Applications import ClustalOmegaCommandline

# Lecture du fichier FASTA contenant la squence de rfrence
sequence_file = "/home/serigne/BIOINFORMATICS/PROJETS/Analyse_Gene_p53/p53.fasta"  
sequences = list(SeqIO.parse(sequence_file, "fasta"))

# Slection de la squence de rfrence (premire squence dans le fichier)
reference_sequence = sequences[0]

# Recherche BLAST pour des squences similaires  la squence de rfrence
result_handle = NCBIWWW.qblast("blastn", "nr", reference_sequence.seq)

# Enregistrement du rsultat BLAST dans un fichier XML
with open("blast_result.xml", "w") as out_handle:
    out_handle.write(result_handle.read())

# Parsing du rsultat BLAST
blast_records = NCBIXML.parse(open("blast_result.xml"))

# Rcupration des squences similaires
similar_sequences = []
for blast_record in blast_records:
    for alignment in blast_record.alignments:
        similar_sequences.append(alignment.title)

# Enregistrement des squences similaires dans un fichier FASTA
with open("similar_sequences.fasta", "w") as out_handle:
    for sequence in similar_sequences:
        out_handle.write(">" + sequence + "\n" + sequence + "\n")

from Bio import AlignIO
import matplotlib.pyplot as plt

# Fichier FASTA contenant l'alignement
fasta_file = "aligned.fasta"

# Lecture de l'alignement  partir du fichier FASTA
alignment = AlignIO.read(fasta_file, "fasta")

# Cration d'une liste des positions dans l'alignement
positions = range(1, alignment.get_alignment_length() + 1)

# Cration d'une liste des squences
sequences = [str(record.seq) for record in alignment]

# Cration de la figure
fig, ax = plt.subplots()

# Affichage de l'alignement
ax.imshow([[ord(aa) for aa in seq] for seq in sequences], cmap='Greys', aspect='auto')

# Configuration de l'apparence du graphique
ax.set_xlabel("Position dans l'alignement")
ax.set_ylabel("Squences")
ax.set_yticks(range(len(sequences)))
ax.set_yticklabels([record.id for record in alignment], fontsize=8)

# Affichage des numros de position
ax2 = ax.twiny()
ax2.set_xlim(ax.get_xlim())
ax2.set_xticks(positions)
ax2.set_xticklabels(positions, fontsize=8)
ax2.set_xlabel("Position")

# Affichage du graphique
plt.show()
